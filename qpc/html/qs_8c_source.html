314 void QS_endRec_(void) {
315  uint8_t *buf = QS_priv_.buf; /* put in a temporary (register) */
316  QSCtr head = QS_priv_.head;
317  QSCtr end = QS_priv_.end;
318  uint8_t b = QS_priv_.chksum;
319  b ^= (uint8_t)0xFFU; /* invert the bits in the checksum */
320 
321  QS_priv_.used += (QSCtr)2; /* 2 bytes about to be added */
322 
323  if ((b != QS_FRAME) && (b != QS_ESC)) {
324  QS_INSERT_BYTE_(b)
325  }
326  else {
327  QS_INSERT_BYTE_(QS_ESC)
328  QS_INSERT_BYTE_(b ^ QS_ESC_XOR)
329  ++QS_priv_.used; /* account for the ESC byte */
330  }
331 
332  QS_INSERT_BYTE_(QS_FRAME) /* do not escape this QS_FRAME */
333 
334  QS_priv_.head = head; /* save the head */
335 
336  /* overrun over the old data? */
337  if (QS_priv_.used > end) {
338  QS_priv_.used = end; /* the whole buffer is used */
339  QS_priv_.tail = head; /* shift the tail to the old data */

646 uint16_t QS_getByte(void) {
647  uint16_t ret;
648  if (QS_priv_.used == (QSCtr)0) {
649  ret = QS_EOD; /* set End-Of-Data */
650  }
651  else {
652  uint8_t const *buf = QS_priv_.buf; /* put in a temporary */
653  QSCtr tail = QS_priv_.tail; /* put in a temporary (register) */
654  ret = (uint16_t)(QS_PTR_AT_(buf, tail)); /* set the byte to return */
655  ++tail; /* advance the tail */
656  if (tail == QS_priv_.end) { /* tail wrap around? */
657  tail = (QSCtr)0;
658  }
659  QS_priv_.tail = tail; /* update the tail */
660  --QS_priv_.used; /* one less byte used */
661  }
662  return ret; /* return the byte or EOD */
663 }

665 /****************************************************************************/
688 uint8_t const *QS_getBlock(uint16_t *pNbytes) {
689  QSCtr used = QS_priv_.used; /* put in a temporary (register) */
690  uint8_t const *buf;
691 
692  /* any bytes used in the ring buffer? */
693  if (used != (QSCtr)0) {
694  QSCtr tail = QS_priv_.tail; /* put in a temporary (register) */
695  QSCtr end = QS_priv_.end; /* put in a temporary (register) */
696  QSCtr n = (QSCtr)(end - tail);
697  if (n > used) {
698  n = used;
699  }
700  if (n > (QSCtr)(*pNbytes)) {
701  n = (QSCtr)(*pNbytes);
702  }
703  *pNbytes = (uint16_t)n; /* n-bytes available */
704  buf = &QS_PTR_AT_(QS_priv_.buf, tail); /* the bytes are at the tail */
705 
706  QS_priv_.used = (QSCtr)(used - n);
707  tail += n;
708  if (tail == end) {
709  tail = (QSCtr)0;
710  }
711  QS_priv_.tail = tail;
712  }
713 
714  else { /* no bytes available */
715  *pNbytes = (uint16_t)0; /* no bytes available right now */
716  buf = (uint8_t *)0; /* no bytes available right now */
717  }
718  return buf;
719 }
